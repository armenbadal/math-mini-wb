\newcount\problemcount \problemcount=0
\def\problem{\vfill\eject%
  \advance\problemcount by1%
  {\bf \the\problemcount.}~}

\newwrite\ansf
\immediate\openout\ansf=\jobname.ans.tex
\def\answer#1{%
  \immediate\write\ansf{\the\problemcount. {\noexpand\it #1}}%
  \immediate\write\ansf{}}

\def\textproblem#1\answer#2{%
  \problem #1
  \medskip\mathgrid
  \medskip\line{Պատ.՝\dotfill\null}
  \answer{#2}}

\newdimen\onerow \onerow5mm
\newdimen\freespace
\newbox\mathcell
\setbox\mathcell=\hbox{\vbox{\hrule width \onerow height 0pt}}
\def\cell{\vrule\copy\mathcell}
\def\mathgrid{%
  \freespace\vsize
  \advance\freespace by -\pagetotal
  \advance\freespace by -4\onerow
  \def\rows{\noalign{\hrule}}
  \loop
    \edef\rows{\rows &&&&&&&&&&&&&&\cr\noalign{\hrule}}
    \advance\freespace by -\onerow
  \ifnum\freespace>0\repeat
  \bigskip\special{color push rgb 0.5 0.5 0.5}
  \centerline{\vbox{\halign{
    \cell##&\cell##&\cell##&\cell##&%
    \cell##&\cell##&\cell##&\cell##&%
    \cell##&\cell##&\cell##&\cell##&%
    \cell##&\cell##&\cell##\vrule height\onerow\cr
    \rows}}}\medskip\special{color pop}}
